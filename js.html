1. Credit Card Validation

<!DOCTYPE html>
<html>

<head>
<title>Credit Card Validation</title>
</head>

<body>

<script>

function validate(){

    var card=document.getElementById("card").value;

    if(card.length==16)
    {
        alert("Valid Card");
    }

    else
    {
        alert("Invalid Card");
    }
}

</script>

Enter Card Number :

<input type="text" id="card">

<br><br>

<button onclick="validate()">
Validate
</button>

</body>
</html>
Output

3. Page View Counter

<!DOCTYPE html>
<html>

<head>
<title>Page Counter</title>
</head>

<body>

<?php

session_start();

if(isset($_SESSION['count']))
{
    $_SESSION['count']++;
}

else
{
    $_SESSION['count']=1;
}

echo "Page Views : ".$_SESSION['count'];

?>

</body>
</html>
Output
